<objective>
Implement BUDGET-03 cost & token roll-up + replay parity. Roll up child cost
into the parent's `totalCost` immediately after `subResult` returns, BEFORE
emitting `sub-run-completed`, so the existing "last cost-bearing event ===
final.cost" invariant survives unchanged. Add `partialCost: CostSummary` to
the `sub-run-failed` event so failed children contribute their real spend to
the roll-up. Add a `parent-rollup-drift` parity check to
`recomputeAccountingFromTrace` for replay-time tamper detection.

totals = parent's own provider calls + sum of children. Roll-up is recursive
across depth.

`SubRunFailedEvent`; `accumulateSubRunCost` helper in `defaults.ts`;
parent-rollup-drift parity in `recomputeAccountingFromTrace`; 8-field

       export function accumulateSubRunCost(events: readonly RunEvent[]): CostSummary {
         let total = emptyCost();
         for (const event of events) {
           if (event.type === "sub-run-completed") {
             total = addCost(total, event.subResult.cost);
           } else if (event.type === "sub-run-failed") {
             total = addCost(total, event.partialCost);

         }
         return total;

       // BUDGET-03: parent-rollup-drift parity check.
       // Parent's recorded accounting (8 fields) must equal:
       //   localOnly + Σ subRunCompleted.subResult.cost + Σ subRunFailed.partialCost
       // Drift indicates someone mutated either the parent's recorded accounting
       // or the embedded child traces without updating the other.
       const subRunTotal = accumulateSubRunCost(trace.events);
       const localOnlyCost: CostSummary = {
         usd: trace.finalOutput.cost.usd - subRunTotal.usd,
         inputTokens: trace.finalOutput.cost.inputTokens - subRunTotal.inputTokens,
         outputTokens: trace.finalOutput.cost.outputTokens - subRunTotal.outputTokens,
         totalTokens: trace.finalOutput.cost.totalTokens - subRunTotal.totalTokens
       };
       const recomposed: CostSummary = {
         usd: localOnlyCost.usd + subRunTotal.usd,
         inputTokens: localOnlyCost.inputTokens + subRunTotal.inputTokens,
         outputTokens: localOnlyCost.outputTokens + subRunTotal.outputTokens,
         totalTokens: localOnlyCost.totalTokens + subRunTotal.totalTokens
       };
       const recomposedAccounting = createRunAccounting({
         tier: trace.tier,
         ...(trace.budget.caps ? { budget: trace.budget.caps } : {}),
         ...(trace.budget.termination ? { termination: trace.budget.termination } : {}),
         cost: recomposed,
         events: trace.events

       const rollupDrift = firstDifferingField(local, recomposedAccounting);
       if (rollupDrift !== null) {
         throw new DogpileError({
           code: "invalid-configuration",

             recomputed: rollupDrift.recomputed
           }
         });
       }
