export async function generateModelTurn(options: GenerateModelTurnOptions): Promise<ModelResponse> {
  const startedAt = new Date().toISOString();
  let response: ModelResponse;

  throwIfAborted(options.request.signal, options.model.id);

  if (!options.model.stream) {
    response = await options.model.generate(options.request);
    throwIfAborted(options.request.signal, options.model.id);
    recordProviderCall(response, startedAt, options);
    return response;
  }
  // ... streaming path; same recordProviderCall call at end ...
}

function recordProviderCall(
  response: ModelResponse,
  startedAt: string,
  options: GenerateModelTurnOptions
): void {
  options.onProviderCall?.({
    kind: "replay-trace-provider-call",
    callId: options.callId,

    startedAt,
    completedAt: new Date().toISOString(),

    role: options.agent.role,
    request: requestForTrace(options.request),
    response
  });
}

```typescript
function synthesizeProviderEvents(
  trace: Trace,
  providerCalls: readonly ReplayTraceProviderCall[]
): readonly RunEvent[] {
  // Build an augmented event array: insert synthesized model-request/model-response
  // pairs from providerCalls immediately before their corresponding agent-turn events.
  // providerCalls[i] corresponds to the i-th agent-turn in trace.events.
  // This is the canonical replay anchor per D-11; trace.events is NOT mutated.

  const agentTurnIndices: number[] = [];
  for (let i = 0; i < trace.events.length; i++) {
    if (trace.events[i]?.type === "agent-turn") {
      agentTurnIndices.push(i);
    }
  }

  // Filter out any pre-existing model-request/model-response events from stored trace
  // (for forward compatibility with traces produced after Phase 6)
  const baseEvents = trace.events.filter(
    (e) => e.type !== "model-request" && e.type !== "model-response"
  );

  // Recompute agent-turn indices in the filtered array
  const filteredTurnIndices: number[] = [];
  for (let i = 0; i < baseEvents.length; i++) {
    if (baseEvents[i]?.type === "agent-turn") {
      filteredTurnIndices.push(i);
    }
  }

  const result: RunEvent[] = [];
  let baseIdx = 0;
  let turnCount = 0;

  for (const event of baseEvents) {
    if (event.type === "agent-turn") {
      const call = providerCalls[turnCount];
      if (call !== undefined) {
        result.push({
          type: "model-request",
          runId: trace.runId,
          callId: call.callId,
          providerId: call.providerId,
          modelId: call.modelId,
          startedAt: call.startedAt,
          agentId: call.agentId,
          role: call.role,
          request: call.request
        });
        result.push({
          type: "model-response",
          runId: trace.runId,
          callId: call.callId,
          providerId: call.providerId,
          modelId: call.modelId,
          startedAt: call.startedAt,
          completedAt: call.completedAt,
          agentId: call.agentId,
          role: call.role,
          response: call.response
        });
      }
      turnCount += 1;
    }
    result.push(event);
  }

  return result;
}
```
