**Create `src/tests/provenance-shape.test.ts`** following the exact pattern from `src/tests/replay-version-skew.test.ts`:

```typescript
import { readFile, writeFile } from "node:fs/promises";
import { existsSync } from "node:fs";
import { join } from "node:path";
import { fileURLToPath } from "node:url";
import { describe, expect, it } from "vitest";
import { run } from "../index.js";
import { createDeterministicModelProvider } from "../internal.js";

const repoRoot = fileURLToPath(new URL("../..", import.meta.url));
const fixturePath = join(repoRoot, "src/tests/fixtures/provenance-event-v1.json");

async function captureProvenanceEvents() {
  const model = createDeterministicModelProvider({ outputs: ["Hello from agent."] });
  const result = await run({
    intent: "Test provenance shape",
    model,
    protocol: { kind: "sequential", maxTurns: 1 }
  });
  return result.eventLog.events.filter(
    (e) => e.type === "model-request" || e.type === "model-response"
  );
}

describe("provenance event shape contract", () => {
  it("bootstraps or verifies the frozen provenance event shape fixture", async () => {
    if (!existsSync(fixturePath)) {
      const events = await captureProvenanceEvents();
      await writeFile(fixturePath, JSON.stringify(events, null, 2) + "\n", "utf8");
    }
    const raw = await readFile(fixturePath, "utf8");
    const saved = JSON.parse(raw) as unknown;
    const live = await captureProvenanceEvents();
    // Compare shape (field names and types) not exact values (timestamps differ per run)
    expect(live).toHaveLength(2);
    expect(live[0]?.type).toBe("model-request");
    expect(live[1]?.type).toBe("model-response");
    // Shape contract: every field present in the fixture must be present in live output
    const savedArray = saved as Record<string, unknown>[];
    expect(Object.keys(live[0] ?? {})).toEqual(Object.keys(savedArray[0] ?? {}));
    expect(Object.keys(live[1] ?? {})).toEqual(Object.keys(savedArray[1] ?? {}));
  });
});
```

Note: The fixture bootstrap runs on first execution, writing `provenance-event-v1.json`. After it runs, commit the file. The shape comparison (field names, not values) is the stable contract that catches future shape changes. Exact timestamp values are not compared because they differ per run.

Important: Run `pnpm vitest run src/tests/provenance-shape.test.ts` once after Plans 02-04 are complete to bootstrap the fixture. The fixture file will be created automatically.
