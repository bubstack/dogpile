```typescript
import type { RunResult } from "../types.js";
import type { BudgetStopEvent, FinalEvent, RunEvent, SubRunCompletedEvent, TurnEvent } from "../types/events.js";

/**
 * A point-in-time snapshot of counters for a completed run or sub-run.
 *
 * Own counters (inputTokens, outputTokens, costUsd, turns) reflect only direct
 * work performed by this run, excluding nested child sub-run contributions.
 * Total counters (totalInputTokens, totalOutputTokens, totalCostUsd) include
 * the full subtree.
 */
export interface RunMetricsSnapshot {
  readonly outcome: "completed" | "budget-stopped" | "aborted";
  readonly inputTokens: number;
  readonly outputTokens: number;
  readonly costUsd: number;
  readonly totalInputTokens: number;
  readonly totalOutputTokens: number;
  readonly totalCostUsd: number;
  readonly turns: number;
  readonly durationMs: number;
}

/**
 * Callback object for receiving named counters at run and sub-run completion.
 *
 * ## Firing semantics
 *
 * `onRunComplete` fires once per top-level run when the run reaches a terminal
 * state (completed, budget-stopped, or aborted). It does NOT fire for sub-runs
 * dispatched by a coordinator — those are covered by `onSubRunComplete`.
 *
 * `onSubRunComplete` fires from the parent coordinator's emit closure each time
 * a `sub-run-completed` event is seen. It carries a snapshot from the parent's
 * perspective (own cost = child total minus the child's own nested children).
 *
 * ## Error isolation
 *
 * Throwing or async hooks never propagate into the run result. Async hooks are
 * fire-and-forget: errors are routed to `logger.error` (or `console.error` if
 * no logger is configured). Run completion is never delayed by hook latency.
 */
export interface MetricsHook {
  readonly onRunComplete?: (snapshot: RunMetricsSnapshot) => void | Promise<void>;
  readonly onSubRunComplete?: (snapshot: RunMetricsSnapshot) => void | Promise<void>;
}

// Helper: extract nested sub-run costs from trace events.
function nestedCosts(events: RunResult["trace"]["events"]): {
  costUsd: number;
  inputTokens: number;
  outputTokens: number;
} {
  const nestedSubRunEvents = events.filter(
    (e): e is SubRunCompletedEvent => e.type === "sub-run-completed"
  );
  return {
    costUsd: nestedSubRunEvents.reduce((sum, e) => sum + e.subResult.cost.usd, 0),
    inputTokens: nestedSubRunEvents.reduce((sum, e) => sum + e.subResult.cost.inputTokens, 0),
    outputTokens: nestedSubRunEvents.reduce((sum, e) => sum + e.subResult.cost.outputTokens, 0),
  };
}

/**
 * Build a RunMetricsSnapshot for a top-level run result.
 * Called after runNonStreamingProtocol or the streaming execute() resolves.
 *
 * Own cost = result.cost (total) minus nested sub-run costs, matching the same
 * derivation used by buildSubRunSnapshot. Using finalEvent.cost or lastTurnCost
 * as an own-cost proxy is incorrect for coordinator runs because those fields
 * carry cumulative totals — not direct-work-only values.
 */
export function buildRunSnapshot(result: RunResult, durationMs: number): RunMetricsSnapshot {
  const events = result.trace.events;
  const budgetStopEvent = events.find((e): e is BudgetStopEvent => e.type === "budget-stop");
  const finalEvent = events.find((e): e is FinalEvent => e.type === "final");
  const outcome: RunMetricsSnapshot["outcome"] = budgetStopEvent
    ? "budget-stopped"
    : finalEvent
      ? "completed"
      : "aborted";

  const turnEvents = events.filter((e): e is TurnEvent => e.type === "agent-turn");
  const turns = turnEvents.length;

  // Total cost: result.cost is already the full subtree rollup from Phase 2
  const totalInputTokens = result.cost.inputTokens;
  const totalOutputTokens = result.cost.outputTokens;
  const totalCostUsd = result.cost.usd;

  // Own cost: subtract nested child sub-run costs from the total.
  // This is the same derivation as buildSubRunSnapshot and is correct for both
  // leaf runs (no nested children → own == total) and coordinator runs.
  const nested = nestedCosts(events);
  const ownCostUsd = Math.max(0, totalCostUsd - nested.costUsd);
  const ownInputTokens = Math.max(0, totalInputTokens - nested.inputTokens);
  const ownOutputTokens = Math.max(0, totalOutputTokens - nested.outputTokens);

  return {
    outcome,
    inputTokens: ownInputTokens,
    outputTokens: ownOutputTokens,
    costUsd: ownCostUsd,
    totalInputTokens,
    totalOutputTokens,
    totalCostUsd,
    turns,
    durationMs
  };
}

/**
 * Build a RunMetricsSnapshot for a completed sub-run (called from parent's emit closure
 * on sub-run-completed events).
 */
export function buildSubRunSnapshot(subResult: RunResult, durationMs: number): RunMetricsSnapshot {
  const events = subResult.trace.events;
  const budgetStopEvent = events.find((e): e is BudgetStopEvent => e.type === "budget-stop");
  const finalEvent = events.find((e): e is FinalEvent => e.type === "final");
  const outcome: RunMetricsSnapshot["outcome"] = budgetStopEvent
    ? "budget-stopped"
    : finalEvent
      ? "completed"
      : "aborted";

  const turnEvents = events.filter((e): e is TurnEvent => e.type === "agent-turn");
  const turns = turnEvents.length;

  // Total cost comes directly from the already-rolled-up subResult.cost (Phase 2 rollup)
  const totalInputTokens = subResult.cost.inputTokens;
  const totalOutputTokens = subResult.cost.outputTokens;
  const totalCostUsd = subResult.cost.usd;

  // Own cost: subtract nested child sub-run costs from the total
  const nested = nestedCosts(events);
  const ownCostUsd = Math.max(0, totalCostUsd - nested.costUsd);
  const ownInputTokens = Math.max(0, totalInputTokens - nested.inputTokens);
  const ownOutputTokens = Math.max(0, totalOutputTokens - nested.outputTokens);

  return {
    outcome,
    inputTokens: ownInputTokens,
    outputTokens: ownOutputTokens,
    costUsd: ownCostUsd,
    totalInputTokens,
    totalOutputTokens,
    totalCostUsd,
    turns,
    durationMs
  };
}

/**
 * Build a partial RunMetricsSnapshot for an aborted run from accumulated mid-run events.
 *
 * Called by engine.ts `finally` blocks (D-10: hook fires for all terminal states including
 * aborted). Unlike buildRunSnapshot, this operates on raw accumulated events rather than
 * a completed RunResult — no cost rollup is available. Partial cost is derived from the
 * last agent-turn event's cumulative cost fields (best available approximation).
 *
 * Always returns outcome: "aborted". Engine callers must ensure this is only called when
 * the run did not complete normally (i.e., when finalResult === undefined in the finally
 * block).
 */
export function buildPartialSnapshot(events: RunEvent[], durationMs: number): RunMetricsSnapshot {
  const turnEvents = events.filter((e): e is TurnEvent => e.type === "agent-turn");
  const turns = turnEvents.length;
  const lastTurnEvent = turnEvents[turnEvents.length - 1] as (TurnEvent & { cost?: { usd?: number; inputTokens?: number; outputTokens?: number } }) | undefined;
  const lastCost = lastTurnEvent?.cost;
  const partialInputTokens = lastCost?.inputTokens ?? 0;
  const partialOutputTokens = lastCost?.outputTokens ?? 0;
  const partialCostUsd = lastCost?.usd ?? 0;

  return {
    outcome: "aborted",
    inputTokens: partialInputTokens,
    outputTokens: partialOutputTokens,
    costUsd: partialCostUsd,
    totalInputTokens: partialInputTokens,
    totalOutputTokens: partialOutputTokens,
    totalCostUsd: partialCostUsd,
    turns,
    durationMs
  };
}
```
